const express = require('express'); const router = express.Router(); const db = require('../db'); const authenticateToken = require('../middleware/auth'); // üîê POST /api/questions ‚Äî Submit a question router.post('/', authenticateToken, async (req, res) => { const { topic, question } = req.body; console.log('üì• Incoming question submission:'); console.log('üß† Token user:', req.user); console.log('üìö Topic:', topic); console.log('üí¨ Question:', question); if (!topic || !question) { return res.status(400).json({ error: 'Both topic and question are required.' }); } try { await db.execute( 'INSERT INTO questions (user_id, topic, question) VALUES (?, ?, ?)', [req.user.id, topic || null, question || null] ); res.json({ message: 'Question submitted successfully!' }); } catch (err) { console.error('‚ùå DB INSERT ERROR:', err.message); res.status(500).json({ error: 'Failed to submit question.' }); } }); // üîê GET /api/questions/mine ‚Äî Get only current user‚Äôs questions router.get('/mine', authenticateToken, async (req, res) => { try { const [rows] = await db.execute( 'SELECT * FROM questions WHERE user_id = ? ORDER BY id DESC', [req.user.id] ); res.json(rows); } catch (err) { console.error('‚ùå Mine Error:', err.message); res.status(500).json({ error: 'Failed to fetch your questions.' }); } }); // üîê GET /api/questions/all ‚Äî Get all questions router.get('/all', authenticateToken, async (req, res) => { try { const [rows] = await db.execute( 'SELECT * FROM questions ORDER BY id DESC' ); res.json(rows); } catch (err) { console.error('‚ùå All Error:', err.message); res.status(500).json({ error: 'Failed to fetch questions.' }); } }); // üîê GET /api/questions/others ‚Äî Get questions from other users router.get('/others', authenticateToken, async (req, res) => { try { const [rows] = await db.execute( 'SELECT * FROM questions WHERE user_id != ? ORDER BY id DESC', [req.user.id] ); res.json(rows); } catch (err) { console.error('‚ùå Others Error:', err.message); res.status(500).json({ error: 'Failed to fetch others\' questions.' }); } }); module.exports = router;
